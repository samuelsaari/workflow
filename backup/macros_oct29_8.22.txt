Sub A_TEST_SHORTCUTS()

' Backlash         \    = '
' Slash     /    = -
' CloseSquareBrace ]    = ¨
CustomizationContext = NormalTemplate
KeyBindings.Add KeyCategory:=wdKeyCategoryMacro, _
    Command:="testi1", KeyCode:=BuildKeyCode(wdKeyControl, wdKeyBackSlash), KeyCode2:=BuildKeyCode(wdKeyControl, wdKeySlash)

' singlequote ' = ä
' semicolon ; = ö
KeyBindings.Add KeyCategory:=wdKeyCategoryMacro, _
    Command:="testi2", KeyCode:=BuildKeyCode(wdKeyControl, wdKeySingleQuote), KeyCode2:=BuildKeyCode(wdKeyControl, wdKeySemiColon)

End Sub


Sub testi1()
'
' testi1 Macro
'
'
    Selection.Font.Bold = wdToggle
End Sub
Sub testi2()
'
' testi2 Macro
'
'
    Selection.Font.Italic = wdToggle
End Sub
